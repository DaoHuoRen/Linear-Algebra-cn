""" SVD
    
        A:  listnumpy shape(m,n)
        topk:  (m,n) -> (m,topk)
        
    
         (m, n)
    hint
        
        1. UDVtopk
        2. Utopk * topk * V^Ttopk(V,V^T)
        
"""
def calc_svd_reconsitution(A, topk = 2):
    u,d,vt = calc_svd(A)
    return np.dot(np.dot(u[:,:topk],np.diag(d[:topk])),vt[:topk,:])
